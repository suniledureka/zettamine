Counter.test.js

test(name of test, callback function)

test('render the text area', callback function - which contains all the testing part)

3 basic things will be done in the callback function of test()
 (a) render the component that we want to test
 (b) try to search the element that we are trying to test
 (c) we check the condition
---> we have rendered, we have query, and we have condition

Counter.test.js

import { render } from "@testing-library/react";
import { Counter } from "../Counter";

test("render the textarea", () => {
    console.log(render(<Counter />));
});

cmd> npm test


import { render } from "@testing-library/react";
import { Counter } from "../Counter";

test("render the textarea", () => {
    const { getByTestId } = render(<Counter />);
    const textArea = getByTestId("textArea");
    expect(textArea).toBeTruthy()  //exists or not 
});

cmd> npm start
